122      @SuppressLint("NewApi")
123      private boolean arePermissionsDenied() {
124  
125          int p = 0;
126          while (p < PERMISSIONS_COUNT) {
127              if (checkSelfPermission(PERMISSIONS[p]) != PackageManager.PERMISSION_GRANTED) {
128                  return true;
129              }
130              p++;
131          }
132  
133          return false;
134      }

294      private void deleteFileOrFolder(File fileOrFolder){
295        if(fileOrFolder.isDirectory()){
296            if(fileOrFolder.list().length == 0){
297                fileOrFolder.delete();
298            }else{
299                String files[] = fileOrFolder.list();
300                for(String tmp:files){
301                    File fileToDelete = new File(fileOrFolder, tmp);
302                    deleteFileOrFolder(fileToDelete);
303                }
304                if(fileOrFolder.list().length==0){
305                    fileOrFolder.delete();
306                }
307            }
308        }else{
309            fileOrFolder.delete();
310        }

317                                             final String[] permissions, final int[] grantResults){
318          super.onRequestPermissionsResult(requestCode, permissions, grantResults);
319          if(requestCode==REQUEST_PERMISSIONS && grantResults.length > 0) {
320            if(arePermissionsDenied()){
321                ((ActivityManager) Objects.requireNonNull(this.getSystemService(ACTIVITY_SERVICE))).clearApplicationUserData();

324                onResume();
